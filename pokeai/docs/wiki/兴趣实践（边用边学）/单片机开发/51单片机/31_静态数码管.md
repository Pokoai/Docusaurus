// 头文件
#include <reg52.h>

// 类型重定义
typedef unsigned char u8;
typedef unsigned int u16;

// 管脚定义
#define DIG P0

// 数码管段选表 code：将数列写到 ROM 中，而非空间极其有限的 RAM 里
u8 code SmgDuanTable = {
    0x3f,0x06,0x5b,0x4f,
    0x66,0x6d,0x7d,0x07,
	0x7f,0x6f,0x77,0x7c,
    0x39,0x5e,0x79,0x71
};

// ms 延时函数
static void delay_ms(u16 n)
{
    n *= 23;
    while (n--);
}


void main()
{   
    DIG = SmgDuanTable(4);
    while (1);
}
